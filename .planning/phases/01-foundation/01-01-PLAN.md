    ```typescript
      trailingSlash: true,
      images: {
        loader: "custom",
        imageSizes: [16, 32, 48, 64, 96, 128, 256, 384],
        deviceSizes: [640, 750, 828, 1080, 1200, 1920, 2048, 3840],
      },
      transpilePackages: ["next-image-export-optimizer"],
      env: {
        nextImageExportOptimizer_imageFolderPath: "public/images",
        nextImageExportOptimizer_exportFolderPath: "out",
        nextImageExportOptimizer_quality: "75",
        nextImageExportOptimizer_storePicturesInWEBP: "true",
        nextImageExportOptimizer_exportFolderName: "nextImageExportOptimizer",
        nextImageExportOptimizer_generateAndUseBlurImages: "true",
        nextImageExportOptimizer_remoteImageCacheTTL: "0",
      },
    };

    export default nextConfig;
    ```

    CRITICAL WARNINGS to verify after writing:
    - `output: "export"` must be present — this is what enables static export

    - Once `loader: "custom"` is set, ALL `<Image>` usages anywhere in the project MUST import from `next-image-export-optimizer`, NOT from `next/image`. Add a comment in the file reminding of this constraint.

    Then run the build to verify the configuration is valid:

    ```bash
    npm run build
    ```
